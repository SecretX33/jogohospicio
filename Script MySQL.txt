drop database if exists projeto_jogo;
create database if not exists projeto_jogo;
use projeto_jogo;

drop table if exists etapa cascade;
drop table if exists save cascade;
drop table if exists jogador cascade;

create table etapa(
	cod INT NOT NULL,				   #Ao trocar o local, sobe para o 100 mais próximo
    descricao VARCHAR(2000) NOT NULL,
	textobt1 VARCHAR(70) NOT NULL,
    textobt2 VARCHAR(70) NOT NULL,
	tipo_quadro INT NOT NULL CHECK(tipo_quadro=0 OR tipo_quadro=1),      #0 = quadro decisão / 1 = quadro avançar   
    ref_op1 INT NOT NULL,              #ref_op1 == referência opção 1
	ref_op2 INT NOT NULL,			   #ref_op2 == referência opção 2
    imp_sanidade1 INT NOT NULL,        #imp == impacto nos status do jogador
    imp_emocional1 INT NOT NULL,
    imp_carisma1 INT NOT NULL,
    imp_coragem1 INT NOT NULL,
	imp_sanidade2 INT NOT NULL,
    imp_emocional2 INT NOT NULL,
    imp_carisma2 INT NOT NULL,
    imp_coragem2 INT NOT NULL,
    timer INT NOT NULL,                                                               #timer > 0 = timer ativo / timer == -1 » timer desativado
	ref_optimer INT NOT NULL CHECK (timer=-1 AND ref_timer=-1 OR ref_timer>0),        #ref_optimer == entra quando o usuario nao seleciona nada durante o tempo do timer / colocar -1 quando timer == -1
    PRIMARY KEY(cod),
    FOREIGN KEY(ref_op1) REFERENCES etapa(cod),
    FOREIGN KEY(ref_op2) REFERENCES etapa(cod),
	FOREIGN KEY(ref_optimer) REFERENCES etapa(cod)
);

create table jogador(
	cod_usuario INT NOT NULL AUTO_INCREMENT,
	usuario VARCHAR(50) NOT NULL,      
    senha VARCHAR(50) NOT NULL,
    apelido VARCHAR(50) NOT NULL,
    tempo_jogo LONG,
    PRIMARY KEY(cod_usuario),
    CONSTRAINT UNIQUE_jogador_usuario UNIQUE(usuario)  #Se assegura que o login escolhido seja unico para que não haja conflitos
);

create table save(
	slot_save INT NOT NULL CHECK (slot_save>=1 AND slot_save<=4),
	cod_usuario INT NOT NULL,
	etapa_atual INT DEFAULT 0,           #etapa_atual == 0 indica que o save ainda não está em uso
    tempo_jogo LONG,           #tempo_jogo == null indica que o save ainda não está em uso
    sanidade INT DEFAULT 0,
    emocional INT DEFAULT 0,
    carisma INT DEFAULT 0,
    coragem INT DEFAULT 0,
    escolhas VARCHAR(1000),
    #PRIMARY KEY(slot_save),
	CONSTRAINT PK_save_cod_usuario FOREIGN KEY(cod_usuario) 
    REFERENCES jogador(cod_usuario)
);


#criando os 4 slots de save automaticamente após o usuario ser criado
DROP TRIGGER IF EXISTS TRIGGER_Jogador_Insert;
DELIMITER $$
CREATE TRIGGER TRIGGER_Jogador_Insert AFTER INSERT
ON jogador FOR EACH ROW
BEGIN
	INSERT INTO save(slot_save,cod_usuario) VALUES(0,NEW.cod_usuario);
	INSERT INTO save(slot_save,cod_usuario) VALUES(1,NEW.cod_usuario);
	INSERT INTO save(slot_save,cod_usuario) VALUES(2,NEW.cod_usuario);
	INSERT INTO save(slot_save,cod_usuario) VALUES(3,NEW.cod_usuario);
END$$
DELIMITER ;

#deletando os 4 slots de save automaticamente antes do usuario ser deletado, para manter a integridade do DB
DROP TRIGGER IF EXISTS TRIGGER_Jogador_Delete;
DELIMITER $$
CREATE TRIGGER TRIGGER_Jogador_Delete BEFORE DELETE
ON jogador FOR EACH ROW
BEGIN
	DELETE FROM save WHERE save.cod_usuario = OLD.cod_usuario;
END$$
DELIMITER ;
